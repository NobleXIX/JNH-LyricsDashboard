Overall sentiment distribution

if (!all(c("positive", "negative") %in% names(sentiment))) {
ggplot() +
theme_void() +
labs(
title = "No sentiment data available yet",
subtitle = "Positive/negative columns not found in sentiment table."
)
} else {
sentiment |>
tidyr::pivot_longer(cols = c(positive, negative), names_to = "sentiment") |>
dplyr::group_by(sentiment) |>
dplyr::summarise(total = sum(value, na.rm = TRUE), .groups = "drop") |>
ggplot(aes(x = sentiment, y = total, fill = sentiment)) +
geom_col() +
labs(
title = "Overall Sentiment (Bing Lexicon)",
x = NULL,
y = "Total word count"
) +
theme_lyrics() +
guides(fill = "none")
}

Sentiment by song

if (!all(c("positive", "negative") %in% names(sentiment))) {
ggplot() +
theme_void() +
labs(
title = "No per-song sentiment available",
subtitle = "Positive/negative columns not found in sentiment table."
)
} else {
sentiment |>
mutate(
net_sentiment = positive - negative,
track_label   = paste(artist, "-", track)
) |>
tidyr::pivot_longer(cols = c(positive, negative),
names_to = "sentiment",
values_to = "count") |>
ggplot(aes(x = reorder(track_label, net_sentiment),
y = count,
fill = sentiment)) +
geom_col(position = "dodge") +
coord_flip() +
labs(
title = "Positive vs Negative Words by Song",
x = NULL,
y = "Word count"
) +
theme_lyrics()
}

Top words in an example song

if (nrow(sentiment) == 0) {
ggplot() +
theme_void() +
labs(
title = "No songs with sentiment scores yet",
subtitle = "Run the data build step to generate sample data."
)
} else {
first_artist <- sentiment$artist[1]
first_track  <- sentiment$track[1]

tidy_words |>
dplyr::filter(artist == first_artist, track == first_track) |>
dplyr::count(word, sort = TRUE) |>
dplyr::slice_head(n = 15) |>
ggplot(aes(x = reorder(word, n), y = n)) +
geom_col() +
coord_flip() +
labs(
title = paste("Top Words in", first_track, "by", first_artist),
x = NULL,
y = "Frequency"
) +
theme_lyrics()
}

Songs ranked by net sentiment (visualization)

if (!all(c("positive", "negative") %in% names(sentiment))) {
  ggplot() +
    theme_void() +
    labs(
      title = "No net sentiment available",
      subtitle = "Positive/negative columns not found in sentiment table."
    )
} else {
  sentiment |>
    mutate(
      net_sentiment = positive - negative,
      track_label   = paste(artist, "-", track),
      sentiment_dir = ifelse(net_sentiment >= 0, "More positive", "More negative")
    ) |>
    ggplot(aes(x = reorder(track_label, net_sentiment),
               y = net_sentiment,
               fill = sentiment_dir)) +
    geom_col() +
    coord_flip() +
    labs(
      title = "Songs Ranked by Net Sentiment",
      x = NULL,
      y = "Net sentiment (positive - negative)",
      fill = "Direction"
    ) +
    theme_lyrics()
}
